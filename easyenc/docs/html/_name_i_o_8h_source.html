111 /*
112  Helper macros for creating temporary buffers with an optimization that
113  below a given size (OptimizedSize) is allocated on the stack, and when a
114  larger size is requested it is allocated on the heap.
115 
116  BUFFER_RESET should be called for the same name as BUFFER_INIT
117 */
118 #define BUFFER_INIT( Name, OptimizedSize, Size ) \
119 char Name ## _Raw [ OptimizedSize ]; \
120 char *Name = Name ## _Raw; \
121 if( sizeof(Name ## _Raw) < Size ) \
122  Name = new char[ Size ];\
123 memset( Name, 0, Size );
124 
125 #define BUFFER_RESET( Name ) \
126 do { \
127  if( Name != Name ## _Raw ) \
128  { \
129  delete[] Name; \
130  Name = Name ## _Raw; \
131  } \
132 } while(0);
